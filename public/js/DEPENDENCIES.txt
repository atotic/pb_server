  function getParent($this) {
    var selector = $this.attr('data-target')
      , $parent

    if (!selector) {
      selector = $this.attr('href')
      selector = selector && /#/.test(selector) && selector.replace(/.*(?=#[^\s]*$)/, '') //strip for ie7
      if (selector == '#')
        selector = '';
    }

    $parent = $(selector)
    $parent.length || ($parent = $this.parent())

    return $parent
  }
